const viewer = document.querySelector('#twin-model'); // This makes the 3D model "clickable" viewer.addEventListener('click', (event) => { // This looks through the 3D space to see what your mouse hit const hit = viewer.materialFromPoint(event.clientX, event.clientY); if (hit) { // This creates a simple text box on your screen showing the data const dataFound = hit.name || "No Name Found"; alert("Object Detected! Internal Name: " + dataFound); } else { console.log("You clicked the background."); } });
